                input.parentElement.querySelector('label').classList.replace('text-on-surface-variant', 'text-primary-fixed');
            });
            input.addEventListener('blur', () = {
                input.parentElement.querySelector('label').classList.replace('text-primary-fixed', 'text-on-surface-variant');
