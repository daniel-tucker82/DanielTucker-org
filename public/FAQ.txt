<script>
        function toggleAccordion(element) {
            const group = element.closest('.accordion-group');
            const isOpen = group.classList.contains('accordion-open');
            
            // Close all others
            document.querySelectorAll('.accordion-group').forEach(item => {
                item.classList.remove('accordion-open');
            });

            if (!isOpen) {
                group.classList.add('accordion-open');
            }
        }

        // Add subtle hover tracking effect to glass cards
        document.querySelectorAll('.glass-card').forEach(card => {
            card.addEventListener('mousemove', e => {
                const rect = card.getBoundingClientRect();
                const x = e.clientX - rect.left;
                const y = e.clientY - rect.top;
                card.style.setProperty('--mouse-x', `${x}px`);
                card.style.setProperty('--mouse-y', `${y}px`);
            });
        });
    </script>
